109 def _palette(self, palette: list[tuple[int, int, int]], shift: int) -> None:
110 # load palette
111  
112 i = 0
113 for e in range(i16(self.fp.read(2))):
114 s = self.fp.read(2)
115 i = i + s[0]
116 n = s[1]
117 if n == 0:
118 n = 256
119 s = self.fp.read(n * 3)
120 for n in range(0, len(s), 3):
121 r = s[n] << shift
122 g = s[n + 1] << shift
123 b = s[n + 2] << shift
