25 def spell_check(request):
26 """
27 Implements the TinyMCE 4 spellchecker protocol
28  
29 :param request: Django http request with JSON-RPC payload from TinyMCE 4
30 containing a language code and a text to check for errors.
31 :type request: django.http.request.HttpRequest
32 :return: Django http response containing JSON-RPC payload
33 with spellcheck results for TinyMCE 4
34 :rtype: django.http.JsonResponse
35 """
36 data = json.loads(request.body.decode('utf-8'))
37 output = {'id': data['id']}
38 error = None
39 status = 200
40 try:
41 if data['params']['lang'] not in list_languages():
42 error = 'Missing {0} dictionary!'.format(data['params']['lang'])
43 raise LookupError(error)
44 spell_checker = checker.SpellChecker(data['params']['lang'])
45 spell_checker.set_text(strip_tags(data['params']['text']))
46 output['result'] = {spell_checker.word: spell_checker.suggest()
47 for err in spell_checker}
48 except NameError:
49 error = 'The pyenchant package is not installed!'
50 logger.exception(error)
51 except LookupError:
52 logger.exception(error)
53 except Exception:
54 error = 'Unknown error!'
55 logger.exception(error)
56 if error is not None:
57 output['error'] = error
58 status = 500
59 return JsonResponse(output, status=status)
