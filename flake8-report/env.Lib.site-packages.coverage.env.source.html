115 def debug_info() -> Iterable[tuple[str, Any]]:
116 """Return a list of (name, value) pairs for printing debug information."""
117 info = [
118 (name, value) for name, value in globals().items()
119 if not name.startswith("_") and name not in _UNINTERESTING_GLOBALS
120 ]
121 info += [
122 (name, value) for name, value in PYBEHAVIOR.__dict__.items()
123 if not name.startswith("_")
124 ]
125 return sorted(info)
