24 def line_hash(line: str) -> str:
25 """Produce a hash of a source line for use in the LCOV file."""
26 # The LCOV file format optionally allows each line to be MD5ed as a
27 # fingerprint of the file. This is not a security use. Some security
28 # scanners raise alarms about the use of MD5 here, but it is a false
29 # positive. This is not a security concern.
30 # The unusual encoding of the MD5 hash, as a base64 sequence with the
31 # trailing = signs stripped, is specified by the LCOV file format.
32 hashed = hashlib.md5(line.encode("utf-8"), usedforsecurity=False).digest()
33 return base64.b64encode(hashed).decode("ascii").rstrip("=")

36 def lcov_lines(
37 analysis: Analysis,
38 lines: list[int],
39 source_lines: list[str],
40 outfile: IO[str],
41 ) -> None:
42 """Emit line coverage records for an analyzed file."""
43 hash_suffix = ""
44 for line in lines:
45 if source_lines:
46 hash_suffix = "," + line_hash(source_lines[line-1])
47 # Q: can we get info about the number of times a statement is
48 # executed? If so, that should be recorded here.
49 hit = int(line not in analysis.missing)
50 outfile.write(f"DA:{line},{hit}{hash_suffix}\n")
51  
52 if analysis.numbers.n_statements > 0:
53 outfile.write(f"LF:{analysis.numbers.n_statements}\n")
54 outfile.write(f"LH:{analysis.numbers.n_executed}\n")

57 def lcov_functions(
58 fr: FileReporter,
59 file_analysis: Analysis,
60 outfile: IO[str],
61 ) -> None:
62 """Emit function coverage records for an analyzed file."""
63 # lcov 2.2 introduces a new format for function coverage records.
64 # We continue to generate the old format because we don't know what
65 # version of the lcov tools will be used to read this report.
66  
67 # "and region.lines" below avoids a crash due to a bug in PyPy 3.8
68 # where, for whatever reason, when collecting data in --branch mode,
69 # top-level functions have an empty lines array. Instead we just don't
70 # emit function records for those.
71  
72 # suppressions because of https://github.com/pylint-dev/pylint/issues/9923
73 functions = [

  • E262 Inline comment should start with '# '
74 (min(region.start, min(region.lines)), #pylint: disable=nested-min-max
  • E261 At least two spaces before inline comment
  • E262 Inline comment should start with '# '
75 max(region.start, max(region.lines)), #pylint: disable=nested-min-max
76 region)
77 for region in fr.code_regions()
78 if region.kind == "function" and region.lines
79 ]
80 if not functions:
81 return
82  
83 functions.sort()
84 functions_hit = 0
85 for first_line, last_line, region in functions:
86 # A function counts as having been executed if any of it has been
87 # executed.
88 analysis = file_analysis.narrow(region.lines)
89 hit = int(analysis.numbers.n_executed > 0)

92 outfile.write(f"FN:{first_line},{last_line},{region.name}\n")
93 outfile.write(f"FNDA:{hit},{region.name}\n")
94  
95 outfile.write(f"FNF:{len(functions)}\n")
96 outfile.write(f"FNH:{functions_hit}\n")

99 def lcov_arcs(
100 fr: FileReporter,
101 analysis: Analysis,
102 lines: list[int],
103 outfile: IO[str],
104 ) -> None:
105 """Emit branch coverage records for an analyzed file."""
106 branch_stats = analysis.branch_stats()
107 executed_arcs = analysis.executed_branch_arcs()
108 missing_arcs = analysis.missing_branch_arcs()
109  
110 for line in lines:
111 if line not in branch_stats:
112 continue
113  
114 # This is only one of several possible ways to map our sets of executed
115 # and not-executed arcs to BRDA codes. It seems to produce reasonable
116 # results when fed through genhtml.
117 _, taken = branch_stats[line]
118  
119 if taken == 0:
120 # When _none_ of the out arcs from 'line' were executed,
121 # it can mean the line always raised an exception.
122 assert len(executed_arcs[line]) == 0
123 destinations = [
124 (dst, "-") for dst in missing_arcs[line]
125 ]
126 else:
  • E501 Line too long (80 > 79 characters)
127 # Q: can we get counts of the number of times each arc was executed?
128 # branch_stats has "total" and "taken" counts for each branch,
129 # but it doesn't have "taken" broken down by destination.
130 destinations = [
131 (dst, "1") for dst in executed_arcs[line]
132 ]
133 destinations.extend(
134 (dst, "0") for dst in missing_arcs[line]
135 )
136  
137 # Sort exit arcs after normal arcs. Exit arcs typically come from
138 # an if statement, at the end of a function, with no else clause.
139 # This structure reads like you're jumping to the end of the function
140 # when the conditional expression is false, so it should be presented
141 # as the second alternative for the branch, after the alternative that
142 # enters the if clause.
143 destinations.sort(key=lambda d: (d[0] < 0, d))
144  
145 for dst, hit in destinations:
146 branch = fr.arc_description(line, dst)
147 outfile.write(f"BRDA:{line},0,{branch},{hit}\n")
148  
149 # Summary of the branch coverage.
150 brf = sum(t for t, k in branch_stats.values())
151 brh = brf - sum(t - k for t, k in branch_stats.values())
152 if brf > 0:
153 outfile.write(f"BRF:{brf}\n")
154 outfile.write(f"BRH:{brh}\n")

168 """Renders the full lcov report.
169  
170 `morfs` is a list of modules or filenames
171  
172 outfile is the file object to write the file into.
173 """
174  
175 self.coverage.get_data()
176 outfile = outfile or sys.stdout

178 # ensure file records are sorted by the _relative_ filename, not the full path
179 to_report = [
180 (fr.relative_filename(), fr, analysis)

191 def lcov_file(
192 self,
193 rel_fname: str,
194 fr: FileReporter,
195 analysis: Analysis,
196 outfile: IO[str],
197 ) -> None:
198 """Produces the lcov data for a single file.
199  
200 This currently supports both line and branch coverage,
201 however function coverage is not supported.
202 """
203  
204 if analysis.numbers.n_statements == 0:
205 if self.config.skip_empty:
206 return
207  
208 outfile.write(f"SF:{rel_fname}\n")
209  
210 lines = sorted(analysis.statements)
211 if self.config.lcov_line_checksums:
212 source_lines = fr.source().splitlines()
213 else:
214 source_lines = []
215  
216 lcov_lines(analysis, lines, source_lines, outfile)
217 lcov_functions(fr, analysis, outfile)
218 if analysis.has_arcs:
