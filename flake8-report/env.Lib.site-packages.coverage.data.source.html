29 def line_counts(data: CoverageData, fullpath: bool = False) -> dict[str, int]:
30 """Return a dict summarizing the line coverage data.
31  
32 Keys are based on the file names, and values are the number of executed
33 lines. If `fullpath` is true, then the keys are the full pathnames of
34 the files, otherwise they are the basenames of the files.
35  
36 Returns a dict mapping file names to counts of lines.
37  
38 """
39 summ = {}
40 filename_fn: Callable[[str], str]
41 if fullpath:
42 # pylint: disable=unnecessary-lambda-assignment
  • E731 Do not assign a lambda expression, use a def
43 filename_fn = lambda f: f
44 else:
45 filename_fn = os.path.basename
46 for filename in data.measured_files():
47 lines = data.lines(filename)
48 assert lines is not None
49 summ[filename_fn(filename)] = len(lines)
50 return summ

99 def combine_parallel_data(
100 data: CoverageData,
101 aliases: PathAliases | None = None,
102 data_paths: Iterable[str] | None = None,
103 strict: bool = False,
104 keep: bool = False,
105 message: Callable[[str], None] | None = None,
106 ) -> None:
107 """Combine a number of data files together.
108  
109 `data` is a CoverageData.
110  
111 Treat `data.filename` as a file prefix, and combine the data from all
112 of the data files starting with that prefix plus a dot.
113  
114 If `aliases` is provided, it's a `PathAliases` object that is used to
115 re-map paths to match the local machine's.
116  
117 If `data_paths` is provided, it is a list of directories or files to
118 combine. Directories are searched for files that start with
119 `data.filename` plus dot as a prefix, and those files are combined.
120  
121 If `data_paths` is not provided, then the directory portion of
122 `data.filename` is used as the directory to search for data files.
123  
124 Unless `keep` is True every data file found and combined is then deleted
125 from disk. If a file cannot be read, a warning will be issued, and the
126 file will not be deleted.
127  
128 If `strict` is true, and no files are found to combine, an error is
129 raised.
130  
131 `message` is a function to use for printing messages to the user.
132  
133 """
134 files_to_combine = combinable_files(data.base_filename(), data_paths)
135  
136 if strict and not files_to_combine:
137 raise NoDataError("No data to combine")
138  
139 if aliases is None:
140 map_path = None
141 else:
142 map_path = functools.cache(aliases.map)
143  
144 file_hashes = set()
145 combined_any = False
146  
147 for f in files_to_combine:
148 if f == data.data_filename():
149 # Sometimes we are combining into a file which is one of the
150 # parallel files. Skip that file.
151 if data._debug.should("dataio"):
152 data._debug.write(f"Skipping combining ourself: {f!r}")
153 continue
154  
155 try:
156 rel_file_name = os.path.relpath(f)
157 except ValueError:
158 # ValueError can be raised under Windows when os.getcwd() returns a
159 # folder from a different drive than the drive of f, in which case
160 # we print the original value of f instead of its relative path
161 rel_file_name = f
162  
163 with open(f, "rb") as fobj:
164 hasher = hashlib.new("sha3_256", usedforsecurity=False)
165 hasher.update(fobj.read())
166 sha = hasher.digest()
167 combine_this_one = sha not in file_hashes
168  
169 delete_this_one = not keep
170 if combine_this_one:
171 if data._debug.should("dataio"):
172 data._debug.write(f"Combining data file {f!r}")
173 file_hashes.add(sha)
174 try:
175 new_data = CoverageData(f, debug=data._debug)
176 new_data.read()
177 except CoverageException as exc:
178 if data._warn:
179 # The CoverageException has the file name in it, so just
180 # use the message as the warning.
181 data._warn(str(exc))
182 if message:

183 message(f"Couldn't combine data file {rel_file_name}: {exc}")
184 delete_this_one = False
185 else:
186 data.update(new_data, map_path=map_path)
187 combined_any = True
188 if message:
189 message(f"Combined data file {rel_file_name}")
190 else:
191 if message:
192 message(f"Skipping duplicate data {rel_file_name}")
193  
194 if delete_this_one:
195 if data._debug.should("dataio"):
196 data._debug.write(f"Deleting data file {f!r}")
197 file_be_gone(f)
198  
199 if strict and not combined_any:
200 raise NoDataError("No usable data files")

203 def debug_data_file(filename: str) -> None:
204 """Implementation of 'coverage debug data'."""
205 data = CoverageData(filename)
206 filename = data.data_filename()
207 print(f"path: {filename}")
208 if not os.path.exists(filename):
209 print("No data collected: file doesn't exist")
210 return
211 data.read()
212 print(f"has_arcs: {data.has_arcs()!r}")
213 summary = line_counts(data, fullpath=True)
214 filenames = human_sorted(summary.keys())
215 nfiles = len(filenames)
216 print(f"{nfiles} file{plural(nfiles)}:")
217 for f in filenames:
218 line = f"{f}: {summary[f]} line{plural(summary[f])}"
219 plugin = data.file_tracer(f)
